from PIL import Image, ImageDraw, ImageFont
import random

def make_album_art(title, artist, out="cover.png"):
    size = 1080
    img = Image.new("RGB", (size, size),
                    (random.randint(0,255), random.randint(0,255), random.randint(0,255)))
    d = ImageDraw.Draw(img)

    for _ in range(80):
        x1, y1 = random.randint(0,size), random.randint(0,size)
        x2, y2 = random.randint(0,size), random.randint(0,size)
        color = (random.randint(0,255), random.randint(0,255), random.randint(0,255))
        d.rectangle([x1,y1,x2,y2], outline=color, width=3)

    try:
        font = ImageFont.truetype("arial.ttf", 80)
        font_small = ImageFont.truetype("arial.ttf", 50)
    except:
        font = font_small = ImageFont.load_default()

    tw, th = d.textsize(title, font=font)
    aw, ah = d.textsize(artist, font=font_small)

    d.text(((size-tw)//2, size//2 - th), title, fill="white", font=font)
    d.text(((size-aw)//2, size//2 + 10), artist, fill="white", font=font_small)

    img.save(out)

make_album_art("Your Title", "Your Name")
